Basic usage

Example#1 Spotify usage

<?php

$s = new Spotify("spotify_appkey.key", "user", "pass");

$playlist = $s->getPlaylistByURI("spotify:user:vilhelmk:playlist:6WIjFOw69JvdX2zST3XKTc");
printf("Playlist name: %s\n", $playlist);

$track = $s->getTrackByURI("spotify:track:1X2F3x6WnF6wVCKgYTrJnn");
if (!$track) {
    printf("Could not get track\n");
} else {
    $duration = $track->getDuration();
    printf("Track name: %s - %s [%02d:%02d]\n", $track->getArtist(), $track, $duration / 60, $duration % 60);
}

$album = $s->getAlbumByURI("spotify:album:1WJR8j3Hn2oq0n75nIJTvS");
if (!$album) {
    printf("Could not get album\n");
} else {
    $tracks = $album->getTracks();
    printf("Album name: %s - %s (%d tracks, from %d)\n", $album->getArtist(), $album, sizeof($tracks), $album->getYear());
}

?>
